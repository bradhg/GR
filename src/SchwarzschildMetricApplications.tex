    \begin{luacode*}
        G = 6.67430e-11  -- m^3/(kg s^2)
        c = 299792458    -- m/s

        function schwarzschildRadius(aMass)
            return 2*G*aMass/c^2   -- m
        end

        function schwarzschildFactor(aMass, aRadius)
            return 1 - schwarzschildRadius(aMass)/aRadius  -- dimensionless
        end

        function print_mm(aLengthMeters)
            printf("\\SI{%.2f}{\\milli\\meter}",aLengthMeters * 1000)
        end

        function print_km(aLengthMeters)
            printf("\\SI{%.2f}{\\kilo\\meter}",aLengthMeters / 1000 )
        end


        function tauDotStationary(aMass, aRadius)
            return math.sqrt( schwarzschildFactor(aMass,aRadius ) )
        end


        mSun = 1.989e30  -- kg

        mEarth = 5.97219e24   -- kg
        rEarth = 6.371e6      -- mean radius, m

        r_s_earth =  schwarzschildRadius(mEarth)
        r_s_sun =  schwarzschildRadius(mSun)


        -- Compute Î”Ï„_{12} = ( Ï„dot2 / Ï„dot1 - 1) * Î”Ï„2
        function deltaTau12(tauDot1, tauDot2, deltaTau2)
            return (tauDot1 / tauDot2 - 1) * deltaTau2
        end


    function printAltitudeTableRows(mass, r2, radii, offset, scale)

        local tauDot2      = tauDotStationary(mass, r2)
        local deltaTau2Day  = 86400
        local deltaTau2Year = 365.25 * deltaTau2Day


        for _, loc in ipairs(radii) do
            local name, r1 = loc.name, loc.radius

            local elev = (r1 - offset) / scale
            --local r1         = r2 + elev
            local tauDot1    = tauDotStationary(mass, r1)

            tex.print(string.format(
                "%s & %s & %.16f & ",
                name,
                commas(elev),
                tauDot1
            ))

            if r1 == r2 then
                -- Print blank cells for Î”Ï„ columns
                tex.print(" &   \\\\")
            else

                local dTau12_day  = deltaTau12(tauDot1, tauDot2, deltaTau2Day)
                local dTau12_year = deltaTau12(tauDot1, tauDot2, deltaTau2Year)

                tex.print(string.format(
                    "%s & %s \\\\",
                    formatSecondsAuto(dTau12_day),
                    formatSecondsAuto(dTau12_year)
                ))
            end
        end

        -- Infinity row
        local tauDot1    = 1.0
        local dTau12_day_inf  = deltaTau12(tauDot1, tauDot2, deltaTau2Day)
        local dTau12_year_inf = deltaTau12(tauDot1, tauDot2, deltaTau2Year)

        tex.print(string.format(
            "Infinity & $\\infty$ & %.16f & %s & %s \\\\",
            tauDot1,
            formatSecondsAuto(dTau12_day_inf),
            formatSecondsAuto(dTau12_year_inf)
        ))
    end

    local R = function(n, radius_m) return {name=n, radius=radius_m} end

    function earthTimeDilationRows()
        locations = {
